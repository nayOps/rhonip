Understanding Multi Sim Sample Application

Functions

The sample shows how to use different SIM card functionalities:

  • Activation/Deactivation of a SIM card
  • Activation/Deactivation of DATA
  • Activation/Deactivation of DATA roaming
  • Configuration and activation of an APN
  • Lock/unlock a SIM card PIN
Note: To get CSP multi-sim module worked, Multi-SIM application must be signed with the provided MorphoCustPackTool script (use '--system-sign-apk' command line option).
Then install the signed APK on a device using a Customization Package.

public long[] getSubState(long subId){
    Log.d(TAG,"getSubState()");
    try {
        Class[] cArg = {long.class};
        Method getSIMState = subscriptionManagerClass.getMethod("getSubState",cArg);
        int resultSim = (int) getSIMState.invoke(subscriptionManager,subId);

        Log.d(TAG,"resultSim" + simIdentifier.getSIMSLot(subId) + " : " + resultSim);
        if(resultSim == 1) {
            Toast.makeText(getActivity(), "SIM " + simIdentifier.getSIMSLot(subId) + " is enabled", Toast.LENGTH_SHORT).show();
        }else{
            Toast.makeText(getActivity(), "SIM " + simIdentifier.getSIMSLot(subId) + " is disabled", Toast.LENGTH_SHORT).show();
        }
    } catch (IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
        Log.d(TAG,"catch error : " + e );
    }
    return null;
}

III) Data Roaming

Activate/deactivate DATA roaming

Parameters :
int roaming
0 : Deactivate data roaming
1 : Activate data roaming
long phoneId - ID of Sim reader
0 : Reader 1
1 : Reader 2

public void setDataRoaming(int roaming, long phoneId){
    Log.d(TAG,"setDataRoaming(" + roaming + ", " + phoneId + ")  " + getActivity());
    if(Settings.Global.putInt(getActivity().getContentResolver(), Settings.Global.DATA_ROAMING + phoneId, roaming)){
        if(roaming == 1) {
            Toast.makeText(getActivity(), "Data roaming for SIM " + (phoneId + 1) + " is activated", Toast.LENGTH_LONG).show();
        }else{
            Toast.makeText(getActivity(), "Data roaming for SIM " + (phoneId + 1) + " is deactivated", Toast.LENGTH_LONG).show();
        }
    }else{
        if(roaming == 1) {
            Toast.makeText(getActivity(), "Activation failed", Toast.LENGTH_LONG).show();
        }else{
            Toast.makeText(getActivity(), "Deactivation failed", Toast.LENGTH_LONG).show();
        }
    }
}

IV) Lock management

Description - Enable / disable SIM card lock

Parameters :
Boolean lock
false : unlocked
true : locked
Object phone (Phone reader to allow lock)

//Enable lock of SIM card
private void enableLock(final Boolean lock, Object phone){

    if(phone == simIdentifier.getPhone1()){
        mActiveSIM = SIM1;
    }else {
        mActiveSIM = SIM2;
    }

    try {
        //get iccCard
        Method getIccCardMethod = mPhoneClass.getMethod("getIccCard");
        mIccCard = getIccCardMethod.invoke(phone);

        // Get SIM lock state
        mGetIccLockEnabled = mIccCardProxyClass.getMethod("getIccLockEnabled");
        mIsIccLockEnabled = (Boolean) mGetIccLockEnabled.invoke(mIccCard);
        logg("Lock Enabled : " + mIsIccLockEnabled);

        Class[] args = {boolean.class, String.class, Message.class};
        mSetIccLockedEnabled = mIccCardProxyClass.getMethod("setIccLockEnabled", args);

    }catch(NoSuchMethodException | IllegalAccessException | InvocationTargetException e){
        logg("catch error : " + e.toString() + "        " + e.getMessage() + "        " + e.getCause().getMessage());
    }

    //Create pop-up to ask PIN code
    AlertDialog.Builder confirmPinDialog = new AlertDialog.Builder(getActivity());
    confirmPinDialog.setMessage("Enter your PIN");
    confirmPinDialog.setTitle("PIN Confirmation");
    final EditText editText = new EditText(getActivity());
    confirmPinDialog.setView(editText);
    final Message callback = Message.obtain(mHandler, 100);

    confirmPinDialog.setPositiveButton("OK", new DialogInterface.OnClickListener() {
        @Override
        public void onClick(DialogInterface dialogInterface, int i) {
            try {

                //Enable or disable lock SIM card
                mSetIccLockedEnabled.invoke(mIccCard, lock, editText.getText().toString(), callback);
            }catch(IllegalAccessException | InvocationTargetException e) {
                logg("catch error : " + e.toString() + "        " + e.getMessage() + "        " + e.getCause().getMessage());
            }
        }
    });

    confirmPinDialog.setNegativeButton("CANCEL", new DialogInterface.OnClickListener() {
        @Override
        public void onClick(DialogInterface dialogInterface, int i) {
            dialogInterface.dismiss();
        }
    });

    AlertDialog dialog = confirmPinDialog.create();
    dialog.show();

}

Description - The callback object returns the state of SIM lock by catching it with an handler and getting the message "100"

Parameters :
Message message return by setIccLockedEnabled

//Get return message of setIccLockedEnabled from callback
private Handler mHandler = new Handler() {
    public void handleMessage(Message msg) {
        switch (msg.what) {
            case 100:
                logg("case 100");
                try {
                    Boolean isIccLockEnabled = (Boolean) mGetIccLockEnabled.invoke(mIccCard);
                    logg("Lock Enabled : " + isIccLockEnabled);
                    if(isIccLockEnabled)
                        Toast.makeText(getActivity(),"SIM " + mActiveSIM + " is locked", Toast.LENGTH_LONG).show();
                    else
                        Toast.makeText(getActivity(),"SIM " + mActiveSIM + " is not locked", Toast.LENGTH_LONG).show();
                }catch(IllegalAccessException | InvocationTargetException e){
                    logg("catch error : " + e);
                }
                break;
            default:
                logg("case " + msg.what);
                break;
        }
    }
};

//Create an APN then select it
private void addNewAPN(long subId) {
    int id;
    Log.d(TAG,"addNewAPN()");
    int iAuthType, iMmsPort, iPort;
    String strName, strApn, strProxy, strMmsProxy, strUser, strServer, strPassword, strMmsc;
    strName = "MyApnName";
    strApn = "Apn";
    strProxy = strMmsProxy = strUser = strServer = strPassword = strMmsc = "aa";
    iAuthType = iMmsPort = iPort = 1;

    String strProtocol, strRoamingProtocol, strType, strMcc, strMnc;
    strProtocol = strRoamingProtocol = strType = "bb";

    strMcc = simIdentifier.getMMC(subId);
    strMnc = simIdentifier.getMNC(subId);

    ContentValues values = new ContentValues();

    values.put(Telephony.Carriers.NAME,strName);
    values.put(Telephony.Carriers.APN,strApn);
    values.put(Telephony.Carriers.PROXY,strProxy);
    values.put(Telephony.Carriers.PORT,iPort);
    values.put(Telephony.Carriers.MMSPROXY,strMmsProxy);
    values.put(Telephony.Carriers.MMSPORT,iMmsPort);
    values.put(Telephony.Carriers.USER,strUser);
    values.put(Telephony.Carriers.SERVER,strServer);
    values.put(Telephony.Carriers.PASSWORD,strPassword);
    values.put(Telephony.Carriers.MMSC,strMmsc);

    values.put(Telephony.Carriers.AUTH_TYPE, iAuthType);
    values.put(Telephony.Carriers.PROTOCOL, strProtocol);
    values.put(Telephony.Carriers.ROAMING_PROTOCOL, strRoamingProtocol);
    values.put(Telephony.Carriers.TYPE, strType);
    values.put(Telephony.Carriers.MCC, strMcc);
    values.put(Telephony.Carriers.MNC, strMnc);

    values.put(Telephony.Carriers.NUMERIC, strMcc + strMnc);

    values.put("apnnew",0);

    Cursor c = null;
    try{
        Uri newRow = getActivity().getContentResolver().insert(getUri(Telephony.Carriers.CONTENT_URI, subId), values);
        if (newRow != null) {
            Log.d(TAG," URI ========= " + newRow);
            c = getActivity().getContentResolver().query(newRow, null, null, null, null);
            int idIndex = c.getColumnIndex("_id");
            c.moveToFirst();
            id = c.getShort(idIndex);
            Log.d(TAG, "New ID: " + id + ": Inserting new APN succeeded!");
            Toast.makeText(getActivity(),strName + " is created", Toast.LENGTH_SHORT).show();
        }else{
            Log.d(TAG,"newRow is null");
        }
    } catch (Exception e) {
        Log.d(TAG,e.getMessage() + "    " + e.getCause().getMessage());
    }
    assert c != null;
    c.close();

    if(setPreferredApn(strName, subId)){
        Log.d(TAG,"setPreferredApn returned TRUE");
        Toast.makeText(getActivity(),strName + " is activated", Toast.LENGTH_SHORT).show();
    }else{
        Log.d(TAG,"setPreferredApn returned FALSE");
    }
}

//Select an APN
public boolean setPreferredApn(String name, long subId) {
    Log.d(TAG,"setPreferredApn(" + name + ")");
    boolean changed = false;
    String columns[] = new String[] { Telephony.Carriers._ID, Telephony.Carriers.NAME };
    String wargs[] = new String[] {name};
    String sortOrder = null;
    Cursor cur = getActivity().getContentResolver().query(APN_TABLE_URI, columns, Telephony.Carriers.NAME + " LIKE ?", wargs, sortOrder);
    if (cur != null) {
        cur.moveToFirst();
        Log.d(TAG,"cursor name : " + cur.getString(NAME_INDEX));
        Log.d(TAG,"cursor id : " + cur.getString(ID_INDEX));
            if(cur.getString(NAME_INDEX).equals(name)) {

                ContentValues values = new ContentValues();

                values.put("apn_id", cur.getLong(ID_INDEX));
                /*******************         set Preferred APN for the SIM                ************************/
                if (getActivity().getContentResolver().update(getUri(APN_PREFER_URI, subId), values, null, null) == 1) {
                    changed = true;
                }
            }
        cur.close();
    }else{
        Log.d(TAG, "cursor null");
    }
    return changed;
}
