Understanding NFC Sample Application

Near Field Communication (NFC) is a set of standards for smartphones and similar devices to create a radio communication with each others by making them touch or getting them close.
NFC standards cover communications protocols and data exchange formats. They are based on existing radio-frequency identification (RFID) standards including ISO/IEC 14443 and FeliCa. The standards include ISO/IEC 18092 and those defined by the NFC Forum.

NfcAdaptor is the Android-based class implementing NFC operations. To enable any NFC operation, you first have to add the following permissions to the manifest file: 

<uses-feature android:name="android.hardware.nfc" android:required="true">
<uses-permission android:name="android.permission.NFC">

B) New Intent

Description – onNewIntent will be automatically called once a new record is detected. Developers can fetch and parse records as follows:

@Override
public void onNewIntent(Intent intent) {
    String action = intent.getAction();
    if (NfcAdapter.ACTION_TAG_DISCOVERED.equals(action)
            || NfcAdapter.ACTION_TECH_DISCOVERED.equals(action)
            || NfcAdapter.ACTION_NDEF_DISCOVERED.equals(action)) {

        displayDetectedRecords(getNdefMessages(intent));
    }
}
/** Get NDEF message from intent */
private NdefMessage[] getNdefMessages(Intent intent) {
    NdefMessage[] ndefMsgs = null;
    String action = intent.getAction();
    if (NfcAdapter.ACTION_TAG_DISCOVERED.equals(action)
            || NfcAdapter.ACTION_NDEF_DISCOVERED.equals(action)) {
        Parcelable[] rawNdefMsgs = intent.getParcelableArrayExtra(NfcAdapter.EXTRA_NDEF_MESSAGES);
        if (rawNdefMsgs != null) {
            ndefMsgs = new NdefMessage[rawNdefMsgs.length];
            java.lang.System.arraycopy(rawNdefMsgs, 0, ndefMsgs, 0, rawNdefMsgs.length);
        } else {
            // Unknown tag type
            Parcelable tag = intent.getParcelableExtra(NfcAdapter.EXTRA_TAG);

            byte[] id = intent.getByteArrayExtra(NfcAdapter.EXTRA_ID);
            byte[] payload = Utils.dumpTagData(tag).getBytes();

            NdefRecord ndefRecord = new NdefRecord(NdefRecord.TNF_UNKNOWN, new byte[0], id, payload);
            NdefMessage ndefMsg = new NdefMessage(new NdefRecord[] { ndefRecord });
            ndefMsgs = new NdefMessage[] { ndefMsg };
        }
    } else {
        Log.e(TAG, "Unknown intent.");
        finish();
    }
    return ndefMsgs;
}

    * Display data that read by application

    public void displayDetectedRecords(NdefMessage[] ndefMsgs) {
    if (ndefMsgs == null || ndefMsgs.length == 0) {
        return;
    }
    
    textView.setVisibility(View.GONE);
    // Parse the first message in the list
    // Build views for all of the sub records
    Date now = new Date();
    List<ParsedNdefRecord> records = NdefMessageParser.parse(ndefMsgs[0]);
    final int size = records.size();
    for (int i = 0; i < size; i++) {
        TextView timeView = new TextView(this);
        timeView.setText(TIME_FORMAT.format(now));  
        timeView.setPadding(0, 10, 10, 0);            
        mContent.addView(timeView, 0);
        ParsedNdefRecord record = records.get(i);
        mContent.addView(record.getParsedView(this), 1 + i);
        View separator=new View(this);
        separator.setBackgroundColor(Color.GRAY);
        separator.setLayoutParams(new LayoutParams(LayoutParams.MATCH_PARENT, 1));
        mContent.addView(separator,1 + i);
    }
    
}

B) Create Text Record

Description – To create a NdefRecord from a string before the write operation

/** Creating a text record */
private NdefMessage createTextRecord(String text) throws UnsupportedEncodingException {
    String lang       = "en";
    byte[] textBytes  = text.getBytes();
    byte[] langBytes  = lang.getBytes("US-ASCII");
    int    langLength = langBytes.length;
    int    textLength = textBytes.length;
    byte[] payload    = new byte[1 + langLength + textLength];

    // set status byte (see NDEF spec for actual bits)
    payload[0] = (byte) langLength;

    // copy langbytes and textbytes into payload
    System.arraycopy(langBytes, 0, payload, 1,              langLength);
    System.arraycopy(textBytes, 0, payload, 1 + langLength, textLength);

    // Creating RTD_TEXT
    NdefRecord recordNFC = new NdefRecord(NdefRecord.TNF_WELL_KNOWN,  NdefRecord.RTD_TEXT,  new byte[0], payload);

    return new NdefMessage(new NdefRecord[] { recordNFC });
}

private NdefMessage createUriRecord(String uriText) throws UnsupportedEncodingException {
    
        byte[] uriField = uriText.getBytes(Charset.forName("US-ASCII"));

    //prefixes http://www. to the URI
    
        System.arraycopy(uriField, 0, payload, 1, uriField.length); //appends URI to payload

}

E) Write

Description – Through write operation, data can be written on NFC.

Parameters:
NdefRecord record - A NdefRecord is created by either createTextRecord(String text) or createUriRecord(String uriText).
Tag tag - tag set by onNewIntent method (tagToBeWritten ).

/** Used to write Record on tag */
private boolean write(NdefMessage message, Tag tag) {
    int size = message.toByteArray().length;
        try {
            Ndef ndef = Ndef.get(tag);
            if (ndef != null) {
                ndef.connect();
                if (!ndef.isWritable()) {
                    Log.e(TAG, " Is not Writable");
                    return false;
                }
                if (ndef.getMaxSize() < size) {
                    Log.e(TAG, " Not have enough size on card.");
                    return false;
                }
                ndef.writeNdefMessage(message);
                return true;
            } else {
                NdefFormatable format = NdefFormatable.get(tag);
                if (format != null) {
                    try {
                        format.connect();
                        format.format(message); 
                        
                        return true;
                    } catch (IOException e) {
                        Log.e(TAG,e.toString());
                        return false;
                    }
                } else {
                    Log.e(TAG," NdefFormatable is null.");
                    return false;
                }
            }
        } catch (Exception e) {
            Log.e(TAG,e.toString());
            return false;
        }

}
